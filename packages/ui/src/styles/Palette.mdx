import { useEffect, useState } from 'react';
import { Meta } from '@storybook/addon-docs/blocks';
import './Palette.css';

<Meta title="Guides/Color Palette" />

export const ThemeToggle = () => {
  const [theme, setTheme] = useState('light');
  useEffect(() => {
    document.documentElement.setAttribute('data-theme', theme);
    document.documentElement.setAttribute('data-platform', 'desktop');
  }, [theme]);
  return (
    <div className="theme-toggle">
      {['light', 'dark'].map((mode) => (
        <button
          key={mode}
          type="button"
          className={mode === theme ? 'active' : ''}
          onClick={() => setTheme(mode)}
        >
          {mode}
        </button>
      ))}
    </div>
  );
};

<ThemeToggle />

# Color Palette

Live view of the **primitive** palette (`brand.css`) and the **semantic** color roles
(`themes/light.css` / `themes/dark.css`) described in
[Theming & Customization](/docs/guides-theming---customization--docs). Swatches render the
actual `var(--...)` value, so this page always reflects the current brand.

---

## Primitives

Raw, semantic-agnostic scale. Retune these to rebrand the whole kit — see
[Theming & Customization](/docs/guides-theming---customization--docs#1-global--a-brand--theme).
